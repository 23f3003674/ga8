import pandas as pd
import matplotlib.pyplot as plt
from io import BytesIO
import base64

# Load sample data
data = """employee_id,department,region,performance_score,years_experience,satisfaction_rating
EMP001,Operations,Latin America,67.11,13,4.6
EMP002,R&D,Middle East,86.04,5,5
EMP003,Operations,Asia Pacific,83.26,7,3.2
EMP004,Sales,Europe,89.88,11,4.7
EMP005,Sales,Europe,85.38,7,4.1
EMP006,HR,Asia Pacific,75.20,8,4.3"""

df = pd.read_csv(pd.io.common.StringIO(data))

# Frequency count for HR department
hr_count = df[df['department'] == 'HR'].shape[0]
print("HR Count:", hr_count)

# Create histogram
plt.figure()
df['department'].value_counts().plot(kind='bar')
plt.title('Department Distribution')
plt.xlabel('Department')
plt.ylabel('Count')

# Convert to base64
buf = BytesIO()
plt.savefig(buf, format="png")
buf.seek(0)
img_base64 = base64.b64encode(buf.read()).decode('utf-8')

# Print base64 for HTML
print(img_base64)
